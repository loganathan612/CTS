cd ~/GitDemo

git status

git branch GitWork
git checkout GitWork

echo "<message>Hello from GitWork</message>" > hello.xml
echo "<branch>GitWork</branch>" >> hello.xml
git status

git add hello.xml
git commit -m "Added hello.xml in GitWork"

git checkout master

echo "<message>Hello from Master</message>" > hello.xml
echo "<branch>Master</branch>" >> hello.xml

git add hello.xml
git commit -m "Added hello.xml in master"

git log --oneline --graph --decorate --all

git diff master GitWork

git merge GitWork

notepad hello.xml

git add hello.xml

git commit -m "Resolved merge conflict in hello.xml"

notepad .gitignore

*.bak // added to .gitignore

git add .gitignore
git commit -m "Ignore backup files"

git branch
git branch -d GitWork

git log --oneline --graph --decorate

git push origin master
